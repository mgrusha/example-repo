public class Main {


    public static void main(String[] args) {
        boolean regularClient=true;
        int discount =  discountCalc(62,regularClient) ;
        System.out.println("Discount is " + discount);
    }

    public static int discountCalc(int age, boolean regularClient){
        int discount = 0;
        if (age >= 65) {
            discount = 10;
        } else if (age < 18) {
            discount = 25;
        }
        if (regularClient){
            discount = discount + 5;
        }
        return discount;
    }
}
